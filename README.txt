SQL cần chạy trong Supabase để form lưu khách hàng mới:

create table if not exists public.leads (
  id bigint generated by default as identity primary key,
  full_name text not null,
  phone text not null,
  email text,
  interest text,
  message text,
  source text default 'website',
  status text default 'new',
  created_at timestamp with time zone default now()
);

alter table public.leads enable row level security;

drop policy if exists "public_can_insert_leads" on public.leads;
create policy "public_can_insert_leads"
on public.leads
for insert
to anon
with check (true);

-- Nếu app nhân viên cần đọc leads, tạo policy riêng cho authenticated users:
drop policy if exists "staff_can_read_leads" on public.leads;
create policy "staff_can_read_leads"
on public.leads
for select
to authenticated
using (true);
